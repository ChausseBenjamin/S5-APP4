\begin{frame}
\frametitle{Compression Matrice A — Théorie}
\begin{enumerate}
    \item Obtenir la matrice de covariance
    \begin{enumerate}
        \item Analyze des données (colonnes)
        \item Construction du vecteur moyen
        \item Centrage des données
        \item Transposé de la matrice
        \item Produit matricielle
    \end{enumerate}
    \item Valeurs propres
    \begin{enumerate}
        \item Application de la formule
        \item Déterminant de la matrice
        \item Valeurs $\lambda$ qui donnent $0$
    \end{enumerate}
    \item Vecteurs propres
    \begin{enumerate}
        \item $\Sigma-\lambda I$ pour chaque $\lambda$ trouvé sachant $\Sigma v=\lambda v$
        \item Isolé $x$ et $y$ de chaque lignes (confirmation)
        \item Posé une valeur ($x$ ou $y$)
        \item Résolution pour trouvé la manquante
    \end{enumerate}
\end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Matrice A - Covariance - Centrage des données}
    \begin{equation}
        \Sigma=\mathbb{E}\left[(X-\mu)(X-\mu)^T\right]
    \end{equation}
    \begin{equation}
        X-\mu
    \end{equation}
    \vspace{1cm}
    \begin{equation}
        \begin{bmatrix}4&1\\2&5\end{bmatrix}-\begin{bmatrix}3\\3\end{bmatrix}=\begin{bmatrix}1&-2\\-1&2\end{bmatrix}
    \end{equation}
\end{frame}

\begin{frame}
    \frametitle{Matrice A - Covariance - Produit matricielle}
    \begin{equation}
        \Sigma=X\times X^T
    \end{equation}
    \begin{equation}
        X^T=\begin{bmatrix}1&-1\\-2&2\end{bmatrix}
    \end{equation}
    \vspace{1cm}
    \begin{equation}
        \begin{bmatrix}1&-2\\-1&2\end{bmatrix}
        \times
        \begin{bmatrix}1&-1\\-2&2\end{bmatrix}
        =
        \begin{bmatrix}2&-4\\-4&8\end{bmatrix}
    \end{equation}
    \begin{equation}
        \Sigma=X^T\times X
    \end{equation}
\end{frame}

\begin{frame}
    \frametitle{Matrice A - Valeurs propres}
    \begin{equation}
        \det(\Sigma-\lambda I)=0
    \end{equation}

    \begin{equation}
    \det\left(
    \begin{bmatrix}2&-4\\-4&8\end{bmatrix}
    -\lambda\begin{bmatrix}1&0\\0&1\end{bmatrix}\right)=0
    \end{equation}

    \begin{equation}
        \det\left(\begin{bmatrix}2-\lambda&-4\\-4&8-\lambda\end{bmatrix}\right)=0
    \end{equation}

    \begin{equation}
        \det\left(\begin{bmatrix}a&b\\ c&d\end{bmatrix}\right)=ad-bc
    \end{equation}

    \begin{equation}
        \det=(2-\lambda)(8-\lambda)-(-4)(-4)\rightarrow\lambda^2-10\lambda
    \end{equation}

    \vspace{0.5cm}
